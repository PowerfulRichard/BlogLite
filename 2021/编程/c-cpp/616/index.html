雄伟的城堡

题目描述

在一个群岛上，有一个富可敌国的大富翁。他打算在这个群岛上建造一个最大城堡，也就是群岛上最大的岛屿。

输入

第一行是一个整数T，代表测试数据的组数。每组数据中第一行是两个整数n,m，代表地图的大小。接下来n行每行共m个整数。0代表海洋，1代表陆地。其中T<=50,n,m<=200

输出

共T行，最大的面积。

#include <bits/stdc++.h>
using namespace std;
#define MAXN 205
int n, m;
bool G[MAXN][MAXN];
int idx[MAXN][MAXN];
int iMax;
void dfs(int r, int c, int id, int &cnt) {
    if(r<0 || r>=n || c<0 || c>=m) return;
    if(G[r][c]==0 || idx[r][c]>0) return;
    idx[r][c] = id;
    cnt++;
    dfs(r-1, c, id, cnt);
    dfs(r, c+1, id, cnt);
    dfs(r+1, c, id, cnt);
    dfs(r, c-1, id, cnt);
}
void doo() {
    memset(G, 0, sizeof(G));
    memset(idx, 0, sizeof(idx));
    iMax = 0;
    cin >> n >> m;
    for(int i=0; i<n; i++)
        for(int j=0; j<m; j++)
            cin >> G[i][j];
    int id = 0;
    for(int i=0; i<n; i++)
        for(int j=0; j<m; j++) {
            int cnt = 0;
            if(G[i][j]==1 && idx[i][j]==0) dfs(i, j, ++id, cnt);
            if(cnt > iMax) iMax = cnt;
        }
   
    cout << iMax << endl;
}
int main()
{
   
    int T;
    cin >> T;
    while(T--)
        doo();
    return 0;
}
