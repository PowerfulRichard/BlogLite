using namespace std;
int main() {
    int n2;
    while(cin&gt;&gt;n2) {
        long long n=n2;                 //避免了对long long 的输入输出
        int t=0;
        while(n!=1) {
            if(n%2==1)
            n=3*n+1;
            else
            n=n/2;
            t++;
        }
        cout&lt;&lt;t&lt;&lt;endl;
    }
    return 0;
